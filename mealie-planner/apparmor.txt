#include <tunables/global>

profile mealie_planner flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Outbound TCP/UDP (Mealie instance + DNS)
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,
  network netlink raw,

  # App files — read-only
  /app/  r,
  /app/** r,

  # Persistent data — credentials + SQLite cache
  /data/  r,
  /data/** rw,

  # Python runtime + installed packages
  /usr/local/lib/    r,
  /usr/local/lib/**  rm,
  /usr/local/bin/    r,
  /usr/local/bin/**  rixm,

  # System libraries
  /lib/**    rm,
  /lib64/**  rm,
  /usr/lib/** rm,

  # Temp files (uvicorn/httpx scratch space)
  /tmp/   r,
  /tmp/** rw,

  # Minimal /proc access required by Python runtime
  @{PROC}/ r,
  @{PROC}/@{pid}/  r,
  @{PROC}/@{pid}/fd/    r,
  @{PROC}/@{pid}/maps   r,
  @{PROC}/@{pid}/status r,
  @{PROC}/sys/kernel/ngroups_max r,
  @{PROC}/sys/kernel/cap_last_cap r,
  @{PROC}/sys/net/core/somaxconn  r,

  # Standard devices
  /dev/null    rw,
  /dev/zero     r,
  /dev/urandom  r,
  /dev/random   r,

  # Hard denials
  deny @{PROC}/sys/kernel/** w,
  deny /sys/**  w,
  deny /etc/shadow r,
  deny /root/** rwx,
}
